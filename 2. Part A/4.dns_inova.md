nano named.conf.local
```
```
zone "localhost" {
        type master;
        file "/etc/bind/db.local";
};

zone "127.in-addr.arpa" {
        type master;
        file "/etc/bind/db.127";
};

```
```
#MY FORWARD ZONES
zone "inova.pt" {
        type master;
        file "/var/lib/bind/db.inova.pt";
};

#MY REVERSE ZONES
zone "31.172.in-addr.arpa" {
        type master;
        file "/var/lib/bind/db.31.172";
};
